include(FetchContent)
FetchContent_Declare(
  googletest
  URL https://github.com/google/googletest/archive/03597a01ee50ed33e9dfd640b249b4be3799d395.zip
)
# For Windows: Prevent overriding the parent project's compiler/linker settings
set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)

add_executable(1_13_Tests test.cpp)
target_sources(1_13_Tests PRIVATE "../src/main.cpp")
target_include_directories(1_13_Tests PRIVATE ../src)
target_link_libraries(1_13_Tests PRIVATE gtest gmock gtest_main)
include(GoogleTest)
gtest_discover_tests(1_13_Tests)
